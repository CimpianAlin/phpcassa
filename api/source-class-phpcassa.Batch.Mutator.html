 61:             list($key, $cf, $cols) = $mut_set;

 64:                 $key_muts = $mutations[$key];
 65:             } else {
 66:                 $key_muts = array();
 67:             }
 68: 
 69:             if (isset($key_muts[$cf])) {
 70:                 $cf_muts = $key_muts[$cf];
 71:             } else {
 72:                 $cf_muts = array();
 73:             }

 78:         }
 79:
