 338:     protected function _multiget($keys, $cp, $slice, $cl, $buffsz) {
 339:         $ret = array();
 340: 
 341:         $have_dict = ($this->return_format == self::DICTIONARY_FORMAT);
 342:         $should_serialize = ($this->key_type instanceof Serialized);
 343:         if ($have_dict) {
 344:             if ($should_serialize) {
 345:                 foreach($keys as $key) {
 346:                     $ret[serialize($key)] = null;
 347:                 }
 348:             } else {
 349:                 foreach($keys as $key) {
 350:                     $ret[$key] = null;
 351:                 }
 352:             }
 353:         }
 354: 
 355:         $cl = $this->rcl($cl);
 356: 
 357:         $resp = array();
 358:         if(count($keys) <= $buffsz) {
 359:             $resp = $this->pool->call("multiget_slice",
 360:                 array_map(array($this, "pack_key"), $keys),
 361:                 $cp, $slice, $cl);
 362:         } else {
 363:             $subset_keys = array();
 364:             $i = 0;
 365:             foreach($keys as $key) {
 366:                 $i += 1;
 367:                 $subset_keys[] = $key;
 368:                 if ($i == $buffsz) {
 369:                     $sub_resp = $this->pool->call("multiget_slice",
 370:                         array_map(array($this, "pack_key"), $subset_keys),
 371:                         $cp, $slice, $cl);
 372:                     $subset_keys = array();
 373:                     $i = 0;
 374:                     $resp = $resp + $sub_resp;
 375:                 }
 376:             }
 377:             if (count($subset_keys) != 0) {
 378:                 $sub_resp = $this->pool->call("multiget_slice",
 379:                     array_map(array($this, "pack_key"), $subset_keys),
 380:                     $cp, $slice, $cl);
 381:                 $resp = $resp + $sub_resp;
 382:             }
 383:         }
 384: 
 385:         $non_empty_keys = array();
 386:         foreach($resp as $key => $val) {
 387:             if (count($val) > 0) {
 388:                 $unpacked_key = $this->unpack_key($key, $have_dict);
 389: 
 390:                 if ($have_dict) {
 391:                     $non_empty_keys[$unpacked_key] = 1;
 392:                     $ret[$unpacked_key] = $this->unpack_coscs($val);
 393:                 } else {
 394:                     $ret[] = array($unpacked_key, $this->unpack_coscs($val));
 395:                 }
 396:             }
 397:         }
 398: 
 399:         if ($have_dict) {
 400:             if ($should_serialize) {
 401:                 foreach($keys as $key) {
 402:                     $skey = serialize($key);
 403:                     if (!isset($non_empty_keys[$skey]))
 404:                         unset($ret[$skey]);
 405:                 }
 406:             } else {
 407:                 foreach($keys as $key) {
 408:                     if (!isset($non_empty_keys[$key]))
 409:                         unset($ret[$key]);
 410:                 }
 411:             }
 412:         }
 413: 
 414:         return $ret;
 415:     }
